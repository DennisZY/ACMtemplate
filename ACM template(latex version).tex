\section{Bisection method}
search for $min(b),b\in\{a[k]\geq x\}$
\begin{lstlisting}
while(l<r){
	int mid = (l + r) >> 1;
	if(a[mid] >= x) r = mid;
	else l = mid + 1;
}
return a[l];
\end{lstlisting} 
search for $max(b),b\in\{a[k]\leq x\}$
\begin{lstlisting}
while(l<r){
	int mid = (l + r + 1) >> 1;
	if(a[mid] <= x) l = mid;
	else r = mid - 1;
}
return a[l];
\end{lstlisting}
