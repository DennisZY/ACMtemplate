int dinic::dfs(int u,int minn) {
	if(u==t)return minn;
	int ret=0;
	for(int i = carc[u]; minn&&i; i=nxt[i]) {
		carc[u]=i;
		int v = ver[i];
		if(flow[i]&&d[v]==d[u]+1) {
			int final=dfs(v,min(flow[i],minn));
			if(final>0) {
				flow[i]-=final;
				flow[i^1]+=final;
				minn-=final;
				ret+=final;
			} else d[v]=-1;
		}
	}
	return ret;
}
